int KNAPSACK(int S){
    int size = items.size();
    int DP[size + 1][S + 1];
    for(int i = 0; i <= size; i++){
        for(int curr_s = 0; curr_s <= S; curr_s++){
            if(i == 0 || curr_s == 0)
                DP[i][curr_s] = 0;
            else if(items[i - 1].S <= curr_s){
                DP[i][curr_s] = max( (items[i - 1].V + DP[i-1][curr_s-items[i-1].S]), DP[i-1][curr_s] );
            } else {
                DP[i][curr_s] = DP[i-1][curr_s];
            }
        }
    }
    return DP[size][S];
}
